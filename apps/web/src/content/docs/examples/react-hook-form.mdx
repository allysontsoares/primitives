# React Hook Form + Zod

Learn how to integrate Kairo with React Hook Form and Zod for robust, accessible form validation.

## Key Features
* **Controlled Integration**: Standard React Hook Form `<Controller>` integration.
* **Zod Schema Validation**: Validate required dates, date formats, and range rules.
* **Accessiblity**: Form error messages linked dynamically to form inputs using `aria-invalid` and `aria-describedby`.

## Code Example

Below is a complete, production-grade implementation of a date picker in a React Hook Form:

```tsx
"use client";

import { useForm, Controller } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import * as z from "zod";
import { DatePicker } from "@at5/kairo";

// 1. Define schema
const schema = z.object({
  appointmentDate: z.date({
    required_error: "Appointment date is required.",
    invalid_type_error: "Please select a valid date.",
  }).refine((date) => date >= new Date(), {
    message: "Appointment date must be today or in the future.",
  }),
});

type FormValues = z.infer<typeof schema>;

export function AppointmentForm() {
  // 2. Setup form
  const { handleSubmit, control, formState: { errors, isSubmitting } } = useForm<FormValues>({
    resolver: zodResolver(schema),
  });

  const onSubmit = (data: FormValues) => {
    console.log("Submitted Date:", data.appointmentDate);
  };

  return (
    <form onSubmit={handleSubmit(onSubmit)} className="space-y-4 max-w-sm">
      <div className="flex flex-col gap-1.5 relative">
        <label htmlFor="appointment-input" className="text-sm font-medium text-zinc-300">
          Preferred Date
        </label>
        
        <Controller
          name="appointmentDate"
          control={control}
          render={({ field }) => (
            <DatePicker.Root
              mode="single"
              value={field.value}
              onValueChange={field.onChange}
            >
              <div className="flex items-center gap-2 rounded-xl border border-zinc-900 bg-zinc-950 px-4 py-2.5 text-sm focus-within:ring-1 focus-within:ring-blue-500 focus-within:border-blue-500 transition-all">
                <DatePicker.Input 
                  id="appointment-input"
                  aria-invalid={!!errors.appointmentDate}
                  aria-describedby={errors.appointmentDate ? "appointment-error" : undefined}
                  className="flex-1 bg-transparent text-zinc-100 outline-none font-mono text-[12px]" 
                />
                <DatePicker.Trigger className="text-zinc-500 hover:text-zinc-100">
                  📅
                </DatePicker.Trigger>
              </div>

              <DatePicker.Content portal={true} side="bottom" align="start" sideOffset={6} className="w-[280px] rounded-xl border border-zinc-900 bg-black p-3.5 shadow-2xl backdrop-blur-md">
                <DatePicker.ViewControl className="flex items-center justify-between mb-2">
                  <DatePicker.PrevTrigger className="h-7 w-7 flex items-center justify-center rounded-lg bg-transparent text-zinc-500 hover:bg-zinc-900 hover:text-white" />
                  <DatePicker.ViewTrigger className="text-xs font-semibold text-zinc-350 hover:bg-zinc-900 transition-colors font-display" />
                  <DatePicker.NextTrigger className="h-7 w-7 flex items-center justify-center rounded-lg bg-transparent text-zinc-500 hover:bg-zinc-900 hover:text-white" />
                </DatePicker.ViewControl>

                <DatePicker.View view="day">
                  <DatePicker.Grid header={<DatePicker.WeekDays className="flex w-full mb-1.5 text-zinc-650 font-bold text-[10px] uppercase font-display" />}>
                    {({ weeks }) => (
                      <tbody className="flex flex-col gap-1">
                        {weeks.map((week, wi) => (
                          <tr key={wi} className="flex w-full justify-between mt-0.5">
                            {week.map((day, di) => (
                              <DatePicker.Day 
                                key={di} 
                                date={day} 
                                className="h-8 w-8 flex items-center justify-center rounded-lg text-xs font-mono font-medium text-zinc-350 hover:bg-zinc-900 hover:text-white aria-selected:bg-blue-600 aria-selected:text-white outline-none"
                              />
                            ))}
                          </tr>
                        ))}
                      </tbody>
                    )}
                  </DatePicker.Grid>
                </DatePicker.View>
              </DatePicker.Content>
            </DatePicker.Root>
          )}
        />

        {/* 3. Accessible error messages */}
        {errors.appointmentDate && (
          <p id="appointment-error" role="alert" className="text-xs text-red-400 mt-1">
            {errors.appointmentDate.message}
          </p>
        )}
      </div>

      <button
        type="submit"
        disabled={isSubmitting}
        className="rounded-lg bg-blue-600 px-4 py-2 text-sm font-semibold text-white hover:bg-blue-500 disabled:opacity-50 transition-colors"
      >
        {isSubmitting ? "Booking..." : "Book Appointment"}
      </button>
    </form>
  );
}
```
